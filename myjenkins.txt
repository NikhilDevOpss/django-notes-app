@Library("shared_hello") _       // library inclusion //
pipeline{
    agent any // if using server than use {label "ec2"} //
    
    stages{
        stage("Hello"){
            steps{
                script{
                    hello()  // using shared library //
                }
            }
        }
        stage("Code"){
            steps{
                echo "Cloning The code"
                git url: "https://github.com/NikhilDevOpss/django-notes-app.git", branch:"main"
                echo "Cloned successfully"
            }
        }
        stage("Build"){
            steps{
                echo "Building the code"
                sh "whoami"
                sh "docker build -t notes-app:latest ."
            }
        }
        stage("Push To Docke Hub"){
            steps{
                echo "push image to the docker hub"
                withCredentials([usernamePassword('credentialsId':"dockerhubcred",passwordVariable:"dockerHubPass",usernameVariable:"dockerHubUser")]){
                sh "docker login -u ${env.dockerHubUser} -p ${env.dockerHubPass}"
                sh "docker image tag notes-app:latest ${env.dockerHubUser}/notes-app:latest"
                sh "docker push ${env.dockerHubUser}/notes-app:latest"
                    
                }
            }
        }
        stage("Deploy"){
            steps{
                echo "Deploying the code"
                sh "docker compose up -d"
            }
        }
    }
}
